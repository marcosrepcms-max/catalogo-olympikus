body : p)); const updated = btoa( unescape(encodeURIComponent(JSON.stringify(updatedList, null, 2))) ); const saveRes = await fetch(GITHUB_API, { method: "PUT", headers: { ...githubHeaders, "Content-Type": "application/json" }, body: JSON.stringify({ message: "update pedido", content: updated, sha, branch: env.GITHUB_BRANCH }) }); if (!saveRes.ok) { const txt = await saveRes.text(); throw new Error(`PUT GitHub falhou (${saveRes.status}): ${txt}`); } return new Response(JSON.stringify({ success: true }), { status: 200, headers: corsHeaders }); } return new Response(JSON.stringify({ error: "Not found" }), { status: 404, headers: corsHeaders }); } catch (error) { return new Response( JSON.stringify({ success: false, error: error.message || "Erro interno no Worker" }), { status: 500, headers: corsHeaders } ); } } };
